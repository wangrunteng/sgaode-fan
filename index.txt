    <script>
        // 页面加载完成后执行
        document.addEventListener('DOMContentLoaded', function() {
            // 初始化滚动动画
            initScrollAnimations();
            
            // 初始化产品切换
            initProductTabs();
            
            // 初始化询价功能
            initInquiry();
            
            // 初始化表单提交
            initFormSubmission();
            
            // 加载询价记录
            loadInquiries();
        });
        
        // 滚动动画
        function initScrollAnimations() {
            const fadeElements = document.querySelectorAll('.fade-in');
            
            const observer = new IntersectionObserver((entries) => {
                entries.forEach(entry => {
                    if (entry.isIntersecting) {
                        entry.target.classList.add('visible');
                    }
                });
            }, {
                threshold: 0.1
            });
            
            fadeElements.forEach(element => {
                observer.observe(element);
            });
        }
        
        // 产品标签切换
        function initProductTabs() {
            const tabs = document.querySelectorAll('.product-tab');
            const contents = document.querySelectorAll('.product-content');
            
            tabs.forEach(tab => {
                tab.addEventListener('click', function() {
                    // 移除所有标签的active类
                    tabs.forEach(t => t.classList.remove('active', 'border-blue-600', 'text-blue-600'));
                    // 为当前标签添加active类
                    this.classList.add('active', 'border-blue-600', 'text-blue-600');
                    
                    // 隐藏所有内容
                    contents.forEach(content => content.classList.add('hidden'));
                    
                    // 显示对应内容
                    const productId = this.getAttribute('data-product');
                    document.getElementById(productId).classList.remove('hidden');
                    
                    // 使用GSAP添加动画效果
                    gsap.from(`#${productId}`, {
                        duration: 0.5,
                        opacity: 0,
                        y: 20,
                        ease: "power2.out"
                    });
                });
            });
            
            // 产品卡片点击事件
            document.querySelectorAll('.view-details').forEach(button => {
                button.addEventListener('click', function() {
                    const productId = this.getAttribute('data-product');
                    
                    // 找到对应标签并点击
                    document.querySelector(`.product-tab[data-product="${productId}"]`).click();
                    
                    // 滚动到产品详情区域
                    document.getElementById('specs').scrollIntoView({ behavior: 'smooth' });
                });
            });
        }
        
        // 询价功能
        function initInquiry() {
            document.querySelectorAll('.inquiry-btn').forEach(button => {
                button.addEventListener('click', function() {
                    const product = this.getAttribute('data-product');
                    
                    // 设置询价表单中的产品选择
                    document.getElementById('product').value = product;
                    
                    // 滚动到询价表单
                    document.getElementById('inquiry').scrollIntoView({ behavior: 'smooth' });
                    
                    // 高亮显示产品选择框
                    const productSelect = document.getElementById('product');
                    productSelect.focus();
                    
                    // 添加动画效果
                    gsap.fromTo(productSelect, 
                        { backgroundColor: '#fef3c7' },
                        { backgroundColor: '#ffffff', duration: 2, ease: "power2.out" }
                    );
                });
            });
        }
        
        // 表单提交
        function initFormSubmission() {
            const form = document.getElementById('inquiry-form');
            
            form.addEventListener('submit', function(e) {
                e.preventDefault();
                
                // 获取表单数据
                const formData = {
                    name: document.getElementById('name').value,
                    company: document.getElementById('company').value,
                    phone: document.getElementById('phone').value,
                    email: document.getElementById('email').value,
                    product: document.getElementById('product').value,
                    message: document.getElementById('message').value,
                    timestamp: new Date().toLocaleString('zh-CN')
                };
                
                // 保存询价数据到localStorage
                saveInquiry(formData);
                
                // 显示成功消息
                alert('感谢您的询价！我们会尽快与您联系。');
                form.reset();
                
                // 重新加载询价记录
                loadInquiries();
            });
        }
        
        // 保存询价数据到localStorage
        function saveInquiry(inquiryData) {
            let inquiries = JSON.parse(localStorage.getItem('inquiries') || '[]');
            inquiries.unshift(inquiryData); // 添加到数组开头
            localStorage.setItem('inquiries', JSON.stringify(inquiries));
        }
        
        // 加载询价记录
        function loadInquiries() {
            const inquiriesList = document.getElementById('inquiries-list');
            const emptyInquiries = document.getElementById('empty-inquiries');
            const inquiries = JSON.parse(localStorage.getItem('inquiries') || '[]');
            
            // 清空当前列表
            inquiriesList.innerHTML = '';
            
            if (inquiries.length === 0) {
                inquiriesList.appendChild(emptyInquiries);
                emptyInquiries.classList.remove('hidden');
                return;
            }
            
            emptyInquiries.classList.add('hidden');
            
            // 添加询价记录
            inquiries.forEach((inquiry, index) => {
                const inquiryItem = document.createElement('div');
                inquiryItem.className = 'bg-gray-50 p-4 rounded-lg inquiry-item';
                inquiryItem.innerHTML = `
                    <div class="flex justify-between items-start mb-2">
                        <div>
                            <h4 class="font-semibold">${inquiry.name}</h4>
                            <p class="text-sm text-gray-600">${inquiry.company || '未填写公司名称'}</p>
                        </div>
                        <span class="text-sm text-gray-500">${inquiry.timestamp}</span>
                    </div>
                    <div class="grid grid-cols-1 md:grid-cols-2 gap-2 mb-2">
                        <div class="flex items-center">
                            <i class="fas fa-phone text-gray-500 mr-2"></i>
                            <span>${inquiry.phone}</span>
                        </div>
                        <div class="flex items-center">
                            <i class="fas fa-envelope text-gray-500 mr-2"></i>
                            <span>${inquiry.email}</span>
                        </div>
                    </div>
                    <div class="mb-2">
                        <span class="inline-block bg-blue-100 text-blue-800 text-xs px-2 py-1 rounded">${inquiry.product}</span>
                    </div>
                    <div>
                        <p class="text-gray-700">${inquiry.message || '未填写具体需求'}</p>
                    </div>
                `;
                inquiriesList.appendChild(inquiryItem);
                
                // 添加动画效果
                gsap.from(inquiryItem, {
                    duration: 0.5,
                    opacity: 0,
                    y: 20,
                    delay: index * 0.1,
                    ease: "power2.out"
                });
            });
        }
